import java.util.Scanner;

public class MayTinh {
    public static void main(String[] args) {
        char PhepTinh;
        Scanner scanner = new Scanner(System.in);
        boolean kiemTraDauVao = true;
        double a= 0;
        do {
            try {
                System.out.print("Mời bạn nhập số a : ");
                a = scanner.nextDouble();
                kiemTraDauVao = true;
            } catch (Exception e) {
                kiemTraDauVao = false;
                System.out.println("Nhập Lại");
                scanner.nextLine();
            }
        } while (!kiemTraDauVao);
        double b= 0;
        do {
            try {
                System.out.print("Mời bạn nhập số b : ");
                b = scanner.nextDouble();
                kiemTraDauVao = true;
            } catch (Exception e) {
                kiemTraDauVao = false;
                System.out.println("Nhập Lại");
                scanner.nextLine();
            }
        } while (!kiemTraDauVao);
        System.out.println("Mời bạn nhập phép tính : ");
        PhepTinh = scanner.next().charAt(0);
        System.out.print("Phép tính " + a + PhepTinh + b);
        System.out.print(" có kết quả là :");
        switch (PhepTinh) {
            case '+':
                System.out.println(a + b);
                break;
            case '-':
                System.out.println(a - b);
                break;
            case '*':
                System.out.println(a * b);
                break;
            case '/': {
                if (b == 0)
                    System.out.println("Số bị chia phải khác 0.");
                else
                    System.out.println(a + " / " + b + " = " + (float) a / b);

            }
            break;
            default:
                System.out.println("Nhập phép tính không hợp lệ");

        }
    }
}
